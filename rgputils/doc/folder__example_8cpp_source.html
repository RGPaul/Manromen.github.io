56  if (list->size () == 0) {

58  }
59 
60  // print all entries from the folder
61  for (FolderEntry entry : *list) {

63  }
64  }
65  else {
66  std::cout << folder.path () << " couldn't list folder entries."
67  << std::endl;
68  }
69  }

72  }
73 
74  // print folder for app data

80  // print home folder

99  }
