69 // normal print
70 void Log::print (const std::string text, const AnsiSgrFgColor fgcolor,
71  const AnsiSgrBgColor bgcolor)
72 {
73  if (_logLevel >= LoglevelNormal) {
74 
75  _cout_mutex.lock();
76 
77  // use log file if possible
78  if (_hasLogfile) {
79 
80  // open log file
81  std::ofstream file;
82  file.open(_logFilePath, std::ios::app);
83 
84  // write output to file
85  if (file.is_open()) {
86 
87  // add time to file
88  time_t timestamp = time(NULL); // get time now
89  tm *current_time = localtime( &timestamp );
90  file
91  << (current_time->tm_year + 1900) << '-'
92  << (current_time->tm_mon + 1) << '-'
93  << current_time->tm_mday << ' '
94  << current_time->tm_hour << ':'
95  << current_time->tm_min << ':'
96  << current_time->tm_sec << ' ';
97 
98  // append text
99  file << text << std::endl;
100  }
101 
102  // close log file
103  file.close();
104 
105  } else {
106 
107  // TODO: check if terminal supports ansi colors
108  if (_useAnsiSgrCodes)
109  std::cout << createSelectGraphicRenditionCode(fgcolor, bgcolor);
110 
111  // output to stdout
112  std::cout << text << std::endl << std::flush;
113 
114  // reset colors to default
115  if (_useAnsiSgrCodes)
116  std::cout << resetSelectGraphicRenditionCode();
117  }
118 
119  _cout_mutex.unlock();
120  }
121 }
